
Implement Heroku.

------------------------

within your app.js add the following lines of code:


const cors = require('cors') // Place this with other requires (like 'path' and 'express')


const corsOptions = {
    origin: "https://<your_app_name>.herokuapp.com/",
    optionsSuccessStatus: 200
};
app.use(cors(corsOptions));

const options = {
    useUnifiedTopology: true,
    useNewUrlParser: true,
    useCreateIndex: true,
    useFindAndModify: false,
    family: 4
};


const MONGODB_URL = process.env.MONGODB_URL || "mongodb+srv://<username>:<username>@cse341cluster-3dwlw.mongodb.net/test?retryWrites=true&w=majority";

const PORT = process.env.PORT || 3000;

---------

modify your mongoose.connect() function to match this:

mongoose.connect(MONGODB_URL, options)
  .then(result => {
    ... // This should be your user handling code implement following the course videos
    app.listen(PORT);
  })
  .catch(err => {
    console.log(err);
  });

---------

push your application to Heroku master and check if your app works on Heroku.

Heroku logs command: heroku logs --tail
